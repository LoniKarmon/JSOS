# Intel I219-LM (e1000e) NIC Driver Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an Intel I219-LM (e1000e) NIC driver with a NicDriver trait abstraction, priority-based NIC selection (e1000e > RTL8139), and 64-bit DMA support.

**Architecture:** A `NicDriver` trait defines the driver interface. A `Nic` enum wraps concrete drivers and implements smoltcp's `Device` trait via delegation. The e1000e driver uses MMIO, descriptor ring DMA, and supports 64-bit physical addresses. PCI discovery tries e1000e first, falls back to RTL8139.

**Tech Stack:** Rust `no_std`, smoltcp 0.11, x86_64 MMIO, PCI config space, DMA descriptor rings.

```rust
/// Translate a virtual address to its 64-bit physical address for DMA use.
/// Unlike `virt_to_phys`, this does not assert 32-bit address range —
/// suitable for devices that support 64-bit DMA (e.g., e1000e).
pub fn virt_to_phys64(virt: u64) -> u64 {
    let offset = PHYS_MEM_OFFSET.load(Ordering::Relaxed);
    if offset == 0 {
        panic!("virt_to_phys64 called before PHYS_MEM_OFFSET initialized");
    }
    unsafe { walk_page_tables(virt, offset) }
}
```

```rust
/// Trait that every NIC driver must implement.
pub trait NicDriver {
    fn mac_address(&self) -> [u8; 6];
    fn receive_packet(&mut self) -> Option<Vec<u8>>;
    fn transmit_packet(&mut self, data: &[u8]);
    fn capabilities(&self) -> DeviceCapabilities;
}

/// Enum wrapping all supported NIC drivers.
/// Needed because smoltcp's Device trait uses GATs, preventing dyn dispatch.
pub enum Nic {
    Rtl8139(Rtl8139),
}

impl NicDriver for Nic {
    fn mac_address(&self) -> [u8; 6] {
        match self {
            Nic::Rtl8139(nic) => nic.mac_address(),
        }
    }

    fn receive_packet(&mut self) -> Option<Vec<u8>> {
        match self {
            Nic::Rtl8139(nic) => nic.receive_packet(),
        }
    }

    fn transmit_packet(&mut self, data: &[u8]) {
        match self {
            Nic::Rtl8139(nic) => nic.transmit_packet(data),
        }
    }

    fn capabilities(&self) -> DeviceCapabilities {
        match self {
            Nic::Rtl8139(nic) => NicDriver::capabilities(nic),
        }
    }
}

// -- smoltcp Device trait delegation --

pub struct NicRxToken(Vec<u8>);

impl RxToken for NicRxToken {
    fn consume<R, F>(self, f: F) -> R
    where
        F: FnOnce(&mut [u8]) -> R,
    {
        let mut packet = self.0;
        f(&mut packet)
    }
}

pub struct NicTxToken<'a>(&'a mut Nic);

impl<'a> TxToken for NicTxToken<'a> {
    fn consume<R, F>(self, len: usize, f: F) -> R
    where
        F: FnOnce(&mut [u8]) -> R,
    {
        let mut buffer = vec![0; len];
        let result = f(&mut buffer);
        self.0.transmit_packet(&buffer);
        result
    }
}

impl Device for Nic {
    type RxToken<'a> = NicRxToken;
    type TxToken<'a> = NicTxToken<'a>;

    fn receive<'a>(&'a mut self, _timestamp: Instant) -> Option<(Self::RxToken<'a>, Self::TxToken<'a>)> {
        match self.receive_packet() {
            Some(packet) => Some((NicRxToken(packet), NicTxToken(self))),
            None => None,
        }
    }

    fn transmit<'a>(&'a mut self, _timestamp: Instant) -> Option<Self::TxToken<'a>> {
        Some(NicTxToken(self))
    }

    fn capabilities(&self) -> DeviceCapabilities {
        NicDriver::capabilities(self)
    }
}
```

```rust
use alloc::vec::Vec;
use crate::pci::{PciDevice, scan_pci, enable_bus_mastering};
use crate::serial_println;
use crate::memory::PHYS_MEM_OFFSET;
use core::sync::atomic::Ordering;

// Intel vendor ID
const INTEL_VENDOR: u16 = 0x8086;

// I219 device ID variants (Comet Lake and others)
const I219_DEVICE_IDS: &[u16] = &[
    0x0D4C, // I219-LM v10 (Comet Lake — primary target for EliteDesk 800 G6)
    0x0D4D, // I219-V v10
    0x0D4E, // I219-LM v11
    0x0D4F, // I219-V v11
    0x15BB, // I219-LM v6
    0x15BC, // I219-V v6
    0x15BD, // I219-LM v7
    0x15BE, // I219-V v7
    0x0D53, // I219-LM v12
    0x15DF, // I219-LM v8
    0x10D3, // Intel 82574L (QEMU e1000e emulation)
];

// --- Register Offsets (from Intel e1000e datasheet) ---
const REG_CTRL: u32     = 0x0000; // Device Control
const REG_STATUS: u32   = 0x0008; // Device Status
const REG_EERD: u32     = 0x0014; // EEPROM Read
const REG_ICR: u32      = 0x00C0; // Interrupt Cause Read
const REG_IMS: u32      = 0x00D0; // Interrupt Mask Set
const REG_IMC: u32      = 0x00D8; // Interrupt Mask Clear
const REG_RCTL: u32     = 0x0100; // Receive Control
const REG_TCTL: u32     = 0x0400; // Transmit Control
const REG_RDBAL: u32    = 0x2800; // RX Descriptor Base Low
const REG_RDBAH: u32    = 0x2804; // RX Descriptor Base High
const REG_RDLEN: u32    = 0x2808; // RX Descriptor Length
const REG_RDH: u32      = 0x2810; // RX Descriptor Head
const REG_RDT: u32      = 0x2818; // RX Descriptor Tail
const REG_TDBAL: u32    = 0x3800; // TX Descriptor Base Low
const REG_TDBAH: u32    = 0x3804; // TX Descriptor Base High
const REG_TDLEN: u32    = 0x3808; // TX Descriptor Length
const REG_TDH: u32      = 0x3810; // TX Descriptor Head
const REG_TDT: u32      = 0x3818; // TX Descriptor Tail
const REG_RAL: u32      = 0x5400; // Receive Address Low
const REG_RAH: u32      = 0x5404; // Receive Address High

// CTRL register bits
const CTRL_RST: u32     = 1 << 26; // Device Reset
const CTRL_SLU: u32     = 1 << 6;  // Set Link Up
const CTRL_ASDE: u32    = 1 << 5;  // Auto-Speed Detection Enable

// RCTL register bits
const RCTL_EN: u32      = 1 << 1;  // Receiver Enable
const RCTL_SBP: u32     = 1 << 2;  // Store Bad Packets
const RCTL_UPE: u32     = 1 << 3;  // Unicast Promiscuous
const RCTL_MPE: u32     = 1 << 4;  // Multicast Promiscuous
const RCTL_BAM: u32     = 1 << 15; // Broadcast Accept Mode
const RCTL_BSIZE_2048: u32 = 0;    // Buffer size 2048 (bits 16-17 = 00)
const RCTL_SECRC: u32   = 1 << 26; // Strip Ethernet CRC

// TCTL register bits
const TCTL_EN: u32      = 1 << 1;  // Transmitter Enable
const TCTL_PSP: u32     = 1 << 3;  // Pad Short Packets
const TCTL_CT_SHIFT: u32 = 4;      // Collision Threshold shift
const TCTL_COLD_SHIFT: u32 = 12;   // Collision Distance shift

// TX command bits
const TX_CMD_EOP: u8    = 1 << 0;  // End of Packet
const TX_CMD_IFCS: u8   = 1 << 1;  // Insert FCS/CRC
const TX_CMD_RS: u8     = 1 << 3;  // Report Status

// Descriptor status bits
const DESC_DD: u8       = 1 << 0;  // Descriptor Done
const RX_DESC_EOP: u8   = 1 << 1;  // End of Packet (RX)

// Ring sizes
const NUM_RX_DESC: usize = 256;
const NUM_TX_DESC: usize = 256;
const PACKET_BUF_SIZE: usize = 2048;

// --- Descriptor Types ---

#[repr(C, packed)]
#[derive(Copy, Clone)]
pub struct RxDescriptor {
    pub buffer_addr: u64,
    pub length: u16,
    pub checksum: u16,
    pub status: u8,
    pub errors: u8,
    pub special: u16,
}

#[repr(C, packed)]
#[derive(Copy, Clone)]
pub struct TxDescriptor {
    pub buffer_addr: u64,
    pub length: u16,
    pub cso: u8,
    pub cmd: u8,
    pub status: u8,
    pub css: u8,
    pub special: u16,
}

// --- Static DMA Buffers ---
// Descriptor rings: 256 * 16 = 4096 bytes each (one page)
// Packet buffers: 256 * 2048 = 512KB each

#[repr(align(4096))]
struct DescRingWrapper<const N: usize>([u8; N]);

#[repr(align(4096))]
struct PacketBufferPool<const N: usize>([u8; N]);

static mut RX_DESC_RING: DescRingWrapper<{ NUM_RX_DESC * 16 }> =
    DescRingWrapper([0u8; NUM_RX_DESC * 16]);
static mut TX_DESC_RING: DescRingWrapper<{ NUM_TX_DESC * 16 }> =
    DescRingWrapper([0u8; NUM_TX_DESC * 16]);
static mut RX_PACKET_BUFS: PacketBufferPool<{ NUM_RX_DESC * PACKET_BUF_SIZE }> =
    PacketBufferPool([0u8; NUM_RX_DESC * PACKET_BUF_SIZE]);
static mut TX_PACKET_BUFS: PacketBufferPool<{ NUM_TX_DESC * PACKET_BUF_SIZE }> =
    PacketBufferPool([0u8; NUM_TX_DESC * PACKET_BUF_SIZE]);

// --- E1000e Driver Struct ---

pub struct E1000e {
    mmio_base: u64,       // virtual address of BAR0 MMIO
    mac_address: [u8; 6],
    rx_tail: u16,         // software RX read pointer
    tx_tail: u16,         // software TX write pointer
}

impl E1000e {
    // --- MMIO read/write helpers ---

    fn read_reg(&self, offset: u32) -> u32 {
        unsafe {
            let ptr = (self.mmio_base + offset as u64) as *const u32;
            core::ptr::read_volatile(ptr)
        }
    }

    fn write_reg(&self, offset: u32, value: u32) {
        unsafe {
            let ptr = (self.mmio_base + offset as u64) as *mut u32;
            core::ptr::write_volatile(ptr, value);
        }
    }
}
```

Add to `impl E1000e` in `src/net/e1000e.rs`:

```rust
    pub fn new(device: &PciDevice) -> Option<Self> {
        enable_bus_mastering(device);

        // BAR0 is memory-mapped
        let bar0_raw = device.get_bar_memory_address(0);
        if bar0_raw == 0 {
            serial_println!("E1000e: No valid BAR0 address");
            return None;
        }
        let bar0_phys = bar0_raw as u64;

        // Map BAR0 to virtual address via physical memory offset
        let phys_offset = PHYS_MEM_OFFSET.load(Ordering::Relaxed);
        let mmio_base = phys_offset + bar0_phys;

        let mut nic = E1000e {
            mmio_base,
            mac_address: [0u8; 6],
            rx_tail: 0,
            tx_tail: 0,
        };

        // 1. Reset device
        nic.reset();

        // 2. Disable interrupts (polled mode)
        nic.write_reg(REG_IMC, 0xFFFF_FFFF);
        // Clear pending interrupts
        let _ = nic.read_reg(REG_ICR);

        // 3. Read MAC address
        nic.mac_address = nic.read_mac();
        if nic.mac_address == [0u8; 6] {
            serial_println!("E1000e: MAC from RAL/RAH is zero, trying EEPROM...");
            nic.mac_address = nic.read_mac_eeprom();
        }
        if nic.mac_address == [0u8; 6] {
            serial_println!("E1000e: Failed to read MAC address");
            return None;
        }

        // 4. Initialize descriptor rings
        nic.init_rx();
        nic.init_tx();

        // 5. Enable RX
        nic.write_reg(REG_RCTL,
            RCTL_EN | RCTL_BAM | RCTL_BSIZE_2048 | RCTL_SECRC);

        // 6. Enable TX
        nic.write_reg(REG_TCTL,
            TCTL_EN | TCTL_PSP
            | (15 << TCTL_CT_SHIFT)     // collision threshold
            | (64 << TCTL_COLD_SHIFT)); // collision distance (full duplex)

        // 7. Set link up
        let ctrl = nic.read_reg(REG_CTRL);
        nic.write_reg(REG_CTRL, ctrl | CTRL_SLU | CTRL_ASDE);

        serial_println!("E1000e: Init complete. MAC: {:02X}:{:02X}:{:02X}:{:02X}:{:02X}:{:02X}",
            nic.mac_address[0], nic.mac_address[1], nic.mac_address[2],
            nic.mac_address[3], nic.mac_address[4], nic.mac_address[5]);

        Some(nic)
    }

    fn reset(&self) {
        let ctrl = self.read_reg(REG_CTRL);
        self.write_reg(REG_CTRL, ctrl | CTRL_RST);
        // Wait for reset to complete (RST bit self-clears)
        for _ in 0..10000 {
            if self.read_reg(REG_CTRL) & CTRL_RST == 0 {
                return;
            }
            core::hint::spin_loop();
        }
        serial_println!("E1000e: WARNING — reset did not self-clear");
    }

    fn read_mac(&self) -> [u8; 6] {
        let ral = self.read_reg(REG_RAL);
        let rah = self.read_reg(REG_RAH);
        [
            (ral & 0xFF) as u8,
            ((ral >> 8) & 0xFF) as u8,
            ((ral >> 16) & 0xFF) as u8,
            ((ral >> 24) & 0xFF) as u8,
            (rah & 0xFF) as u8,
            ((rah >> 8) & 0xFF) as u8,
        ]
    }

    fn read_mac_eeprom(&self) -> [u8; 6] {
        let mut mac = [0u8; 6];
        for i in 0..3u32 {
            // Write EEPROM read command: address in bits 8-15, start bit 0
            self.write_reg(REG_EERD, (i << 8) | 1);
            // Wait for done bit (bit 4)
            for _ in 0..10000 {
                let val = self.read_reg(REG_EERD);
                if val & (1 << 4) != 0 {
                    let data = (val >> 16) as u16;
                    mac[(i * 2) as usize] = (data & 0xFF) as u8;
                    mac[(i * 2 + 1) as usize] = (data >> 8) as u8;
                    break;
                }
                core::hint::spin_loop();
            }
        }
        mac
    }

    fn init_rx(&mut self) {
        let rx_descs = unsafe {
            core::slice::from_raw_parts_mut(
                RX_DESC_RING.0.as_mut_ptr() as *mut RxDescriptor,
                NUM_RX_DESC,
            )
        };

        // Initialize each RX descriptor with a pre-allocated buffer
        for i in 0..NUM_RX_DESC {
            let buf_virt = unsafe {
                RX_PACKET_BUFS.0.as_ptr().add(i * PACKET_BUF_SIZE) as u64
            };
            let buf_phys = crate::memory::virt_to_phys64(buf_virt);
            rx_descs[i] = RxDescriptor {
                buffer_addr: buf_phys,
                length: 0,
                checksum: 0,
                status: 0,
                errors: 0,
                special: 0,
            };
        }

        // Program ring base address (64-bit)
        let ring_phys = crate::memory::virt_to_phys64(
            unsafe { RX_DESC_RING.0.as_ptr() as u64 }
        );
        self.write_reg(REG_RDBAL, ring_phys as u32);
        self.write_reg(REG_RDBAH, (ring_phys >> 32) as u32);
        self.write_reg(REG_RDLEN, (NUM_RX_DESC * 16) as u32);
        self.write_reg(REG_RDH, 0);
        self.write_reg(REG_RDT, (NUM_RX_DESC - 1) as u32);
        self.rx_tail = 0;
    }

    fn init_tx(&mut self) {
        let tx_descs = unsafe {
            core::slice::from_raw_parts_mut(
                TX_DESC_RING.0.as_mut_ptr() as *mut TxDescriptor,
                NUM_TX_DESC,
            )
        };

        // Initialize TX descriptors (buffers assigned on transmit)
        for i in 0..NUM_TX_DESC {
            let buf_virt = unsafe {
                TX_PACKET_BUFS.0.as_ptr().add(i * PACKET_BUF_SIZE) as u64
            };
            let buf_phys = crate::memory::virt_to_phys64(buf_virt);
            tx_descs[i] = TxDescriptor {
                buffer_addr: buf_phys,
                length: 0,
                cso: 0,
                cmd: 0,
                status: DESC_DD, // Mark as "done" so first transmit works
                css: 0,
                special: 0,
            };
        }

        let ring_phys = crate::memory::virt_to_phys64(
            unsafe { TX_DESC_RING.0.as_ptr() as u64 }
        );
        self.write_reg(REG_TDBAL, ring_phys as u32);
        self.write_reg(REG_TDBAH, (ring_phys >> 32) as u32);
        self.write_reg(REG_TDLEN, (NUM_TX_DESC * 16) as u32);
        self.write_reg(REG_TDH, 0);
        self.write_reg(REG_TDT, 0);
        self.tx_tail = 0;
    }
```

```rust
    pub fn mac_address(&self) -> [u8; 6] {
        self.mac_address

    pub fn receive_packet(&mut self) -> Option<Vec<u8>> {
        let rx_descs = unsafe {
            core::slice::from_raw_parts_mut(
                RX_DESC_RING.0.as_mut_ptr() as *mut RxDescriptor,
                NUM_RX_DESC,
            )
        };

        let idx = self.rx_tail as usize;
        let desc = &mut rx_descs[idx];

        // Check if hardware has written to this descriptor (DD bit set)
        if desc.status & DESC_DD == 0 {
            return None;
        }

        let length = desc.length as usize;
        if length == 0 || desc.errors != 0 {
            // Reset descriptor and move on
            desc.status = 0;
            self.rx_tail = ((self.rx_tail as usize + 1) % NUM_RX_DESC) as u16;
            self.write_reg(REG_RDT, self.rx_tail as u32);
            return None;
        }

        // Copy packet data from the RX buffer
        let buf_offset = idx * PACKET_BUF_SIZE;
        let mut packet = alloc::vec![0u8; length];
        unsafe {
            let src = RX_PACKET_BUFS.0.as_ptr().add(buf_offset);
            core::ptr::copy_nonoverlapping(src, packet.as_mut_ptr(), length);
        }

        // Reset descriptor for reuse
        desc.status = 0;
        desc.length = 0;
        desc.errors = 0;

        // Advance tail and notify hardware
        self.rx_tail = ((self.rx_tail as usize + 1) % NUM_RX_DESC) as u16;
        self.write_reg(REG_RDT, self.rx_tail as u32);

        Some(packet)
    }

    pub fn transmit_packet(&mut self, payload: &[u8]) {
        let tx_descs = unsafe {
            core::slice::from_raw_parts_mut(
                TX_DESC_RING.0.as_mut_ptr() as *mut TxDescriptor,
                NUM_TX_DESC,
            )
        };

        let idx = self.tx_tail as usize;
        let desc = &mut tx_descs[idx];

        // Wait for previous transmission to complete (DD bit)
        // In polled mode this should always be set unless we're flooding
        if desc.status & DESC_DD == 0 {
            serial_println!("E1000e: TX ring full, dropping packet");
            return;
        }

        // Copy payload into the TX buffer
        let buf_offset = idx * PACKET_BUF_SIZE;
        let len = payload.len().min(PACKET_BUF_SIZE);
        unsafe {
            let dst = TX_PACKET_BUFS.0.as_mut_ptr().add(buf_offset);
            core::ptr::copy_nonoverlapping(payload.as_ptr(), dst, len);
        }

        // Set up descriptor
        desc.length = len as u16;
        desc.cmd = TX_CMD_EOP | TX_CMD_IFCS | TX_CMD_RS;
        desc.status = 0; // Clear DD — hardware will set it when done

        // Advance tail and notify hardware
        self.tx_tail = ((self.tx_tail as usize + 1) % NUM_TX_DESC) as u16;
        self.write_reg(REG_TDT, self.tx_tail as u32);
    }
```

```rust
pub fn init() {
    // Priority: try e1000e first, fall back to RTL8139
    let nic_opt: Option<Nic> = if let Some(nic) = e1000e::init() {
        serial_println!("[INFO] Using Intel e1000e NIC driver");
        Some(Nic::E1000e(nic))
    } else if let Some(nic) = rtl8139::init() {
        serial_println!("[INFO] Using RTL8139 NIC driver");
        Some(Nic::Rtl8139(nic))
    } else {
        None
    };

    if let Some(nic_instance) = nic_opt {
        let mac_addr = nic::NicDriver::mac_address(&nic_instance);
        let hw_addr = HardwareAddress::Ethernet(EthernetAddress(mac_addr));

        let mut config = Config::new(hw_addr);
        config.random_seed = 0x12345678;

        *NIC.lock() = Some(nic_instance);

        let mut nic_guard = NIC.lock();
        let nic = nic_guard.as_mut().unwrap();
        let iface = Interface::new(config, nic, get_timestamp());
        let mut sockets = SocketSet::new(vec![]);

        sockets.add(Dhcpv4Socket::new());
        let dns_servers = [Ipv4Address::new(8, 8, 8, 8).into()];
        sockets.add(DnsSocket::new(&dns_servers, vec![]));

        *IFACE.lock() = Some(iface);
        *SOCKETS.lock() = Some(sockets);

        serial_println!("[INFO] Network stack initialized. DHCP Client starting...");
    } else {
        println!("No NIC found. Network stack disabled.");
    }
}
```
